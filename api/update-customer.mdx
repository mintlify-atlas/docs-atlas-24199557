## Function Signature

```typescript
export async function updateCustomer(
  body: z.infer<typeof customerSchema>,
  customerId: string
): Promise<Stripe.Customer>
```

## Description

Updates customer information in Stripe, including name, email, and address fields. The function automatically filters out null, undefined, and empty string values before sending the update request to Stripe.

## Parameters

<ParamField path="body" type="object" required>
  The customer data to update, validated against `customerSchema`.
  
  <ParamField path="name" type="string | null">
    Customer's full name
  </ParamField>
  
  <ParamField path="email" type="string | null">
    Customer's email address (must be valid email format)
  </ParamField>
  
  <ParamField path="address" type="object">
    Customer's address information
    
    <ParamField path="city" type="string | null">
      City name
    </ParamField>
    
    <ParamField path="country" type="string | null">
      Country code (e.g., "US", "GB")
    </ParamField>
    
    <ParamField path="line1" type="string | null">
      Address line 1 (street address)
    </ParamField>
    
    <ParamField path="line2" type="string | null">
      Address line 2 (apartment, suite, etc.)
    </ParamField>
    
    <ParamField path="postal_code" type="string | null">
      ZIP or postal code
    </ParamField>
    
    <ParamField path="state" type="string | null">
      State or province
    </ParamField>
  </ParamField>
  
  <ParamField path="currency" type="string">
    Currency code in lowercase (e.g., "usd", "eur")
  </ParamField>
</ParamField>

<ParamField path="customerId" type="string" required>
  The Stripe customer ID (e.g., "cus_123456789")
</ParamField>

## Return Value

<ResponseField name="updated" type="Stripe.Customer">
  Returns the updated Stripe Customer object with all current customer data.
</ResponseField>

## Error Handling

Throws an `Error` if:
- The Stripe API request fails
- The customer ID is invalid
- Network connectivity issues occur

Error message: `"Failed to update Stripe customer"`

## Implementation Details

The function includes a `clean` helper that removes null, undefined, and empty string values from objects before sending to Stripe:

```typescript
const clean = (obj: Record<string, any>) =>
  Object.fromEntries(
    Object.entries(obj).filter(
      ([, value]) => value !== null && value !== undefined && value !== ""
    )
  );
```

This ensures only meaningful data is sent to the Stripe API.

## Usage Example

```typescript
import { updateCustomer } from './lib/stripe-ui-kit';

// Update customer with new information
try {
  const updated = await updateCustomer(
    {
      name: "John Doe",
      email: "john.doe@example.com",
      address: {
        line1: "123 Main St",
        line2: "Apt 4B",
        city: "San Francisco",
        state: "CA",
        postal_code: "94105",
        country: "US"
      },
      currency: "usd"
    },
    "cus_123456789"
  );
  
  console.log("Customer updated:", updated);
} catch (error) {
  console.error("Update failed:", error);
}
```

## Partial Updates

You can provide partial data - only non-null/non-empty fields will be updated:

```typescript
// Only update email
await updateCustomer(
  {
    name: null,
    email: "newemail@example.com",

      line1: null,
      line2: null,
      city: null,
      state: null,
      postal_code: null,
      country: null
    },
    currency: "usd"
  },
  "cus_123456789"
);
```

## Source Code

```typescript
export async function updateCustomer(
  body: z.infer<typeof customerSchema>,
  customerId: string
) {
  try {
    const clean = (obj: Record<string, any>) =>
      Object.fromEntries(
        Object.entries(obj).filter(
          ([, value]) => value !== null && value !== undefined && value !== ""
        )
      );

    const updated = await stripe.customers.update(customerId, {
      name: body.name || undefined,
      email: body.email || undefined,
      address: clean(body.address),
    });

    return updated;
  } catch (error) {
    console.error("Stripe update failed:", error);
    throw new Error("Failed to update Stripe customer");
  }
}
```

## Schema Definition

```typescript
export const customerSchema = z.object({
  address: z.object({
    city: z.string().nullable(),
    country: z.string().nullable(),
    line1: z.string().nullable(),
    line2: z.string().nullable(),
    postal_code: z.string().nullable(),
    state: z.string().nullable(),
  }),
  name: z.string().nullable(),
  email: z.string().email().nullable(),
  currency: z.string().toLowerCase(),
});
```

<Note>
This function is server-side only and requires the `STRIPE_SECRET_KEY` environment variable.
</Note>
